Widgets Fundamentales

Ventana Gráfica (Tk)

El primer paso para cualquier aplicación de Tkinter es crear la ventana principal. Esto se hace instanciando la clase Tk(). El método mainloop() mantiene la ventana abierta y a la espera de eventos.


# Importamos la librería
from tkinter import *

# Creamos la ventana principal (raíz)
root = Tk()

# Asignamos un título a la ventana
root.title("Mi Primera Ventana")

# Iniciamos el bucle principal de la aplicación
root.mainloop()

Frame

Un Frame es un contenedor rectangular invisible que se utiliza para agrupar y organizar otros widgets.


from tkinter import *
root = Tk()
root.title("Ejemplo de Frame")

marco_principal = Frame(root)
marco_principal.pack()

marco_principal.config(width="400", height="300")
marco_principal.config(bg="lightblue")

root.mainloop()

Label

El widget Label se utiliza para mostrar texto o imágenes estáticas.


from tkinter import *
root = Tk()
root.title("Ejemplo de Label")

etiqueta = Label(root, text="¡Hola, mundo Tkinter!")
etiqueta.pack(pady=20, padx=20)

root.mainloop()

Button

Button ejecuta una acción vía el parámetro command.


from tkinter import *

root = Tk()
root.title("Ejemplo de Botón")

def accion_boton():
    print("¡El botón fue presionado!")
    etiqueta.config(text="¡Botón presionado!")

boton = Button(root, text="Presióname", command=accion_boton)
boton.pack(pady=10)

etiqueta = Label(root, text="Esperando acción...")
etiqueta.pack(pady=10)

root.mainloop()

Entry

Campo de texto de una sola línea. Para contraseña: show="*".


from tkinter import *
root = Tk()
root.title("Ejemplo de Entry")

entrada_nombre = Entry(root, width=30)
entrada_nombre.pack(pady=5, padx=10)
entrada_nombre.insert(0, "Escribe tu nombre aquí...")

entrada_pass = Entry(root, width=30, show="*")
entrada_pass.pack(pady=5, padx=10)

def saludar():
    nombre = entrada_nombre.get()
    resultado.config(text=f"Hola, {nombre}")

Button(root, text="Saludar", command=saludar).pack(pady=10)
resultado = Label(root, text="")
resultado.pack()

root.mainloop()

Radiobutton

Seleccionar una única opción de un conjunto mediante una variable de control.


from tkinter import *
root = Tk()
root.title("Ejemplo de Radiobutton")

opcion = IntVar()
opcion.set(1)

def mostrar_opcion():
    resultado.config(text=f"Opción seleccionada: {opcion.get()}")

Label(root, text="Elige una opción:").pack(anchor=W)
Radiobutton(root, text="Opción 1", variable=opcion, value=1).pack(anchor=W)
Radiobutton(root, text="Opción 2", variable=opcion, value=2).pack(anchor=W)
Radiobutton(root, text="Opción 3", variable=opcion, value=3).pack(anchor=W)

Button(root, text="Confirmar", command=mostrar_opcion).pack(pady=10)
resultado = Label(root, text="")
resultado.pack()

root.mainloop()

Checkbutton

Casillas de verificación con su propia variable de control.


from tkinter import *
root = Tk()
root.title("Ejemplo de Checkbutton")

acepta = BooleanVar()

def verificar():
    if acepta.get():
        resultado.config(text="Términos aceptados.")
    else:
        resultado.config(text="Debes aceptar los términos.")

check = Checkbutton(root, text="Acepto los términos y condiciones", variable=acepta)
check.pack(pady=10)

Button(root, text="Verificar", command=verificar).pack()
resultado = Label(root, text="")
resultado.pack()

root.mainloop()

LabelFrame

Similar a Frame pero con borde y título.


from tkinter import *
root = Tk()
root.title("Ejemplo de LabelFrame")

login_frame = LabelFrame(root, text="Inicio de Sesión", padx=10, pady=10)
login_frame.pack(pady=20, padx=20)

Label(login_frame, text="Usuario:").grid(row=0, column=0, sticky="w", pady=2)
Entry(login_frame).grid(row=0, column=1, pady=2)
Label(login_frame, text="Contraseña:").grid(row=1, column=0, sticky="w", pady=2)
Entry(login_frame, show="*").grid(row=1, column=1, pady=2)
Button(login_frame, text="Entrar").grid(row=2, column=0, columnspan=2, pady=10)

root.mainloop()
