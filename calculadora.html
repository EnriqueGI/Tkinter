Ejemplo Práctico: Calculadora

Este ejemplo combina varios widgets (Entry, Button) y el gestor grid() para crear la interfaz de una calculadora.

Código Completo


from tkinter import *

root = Tk()
root.title("Calculadora")
root.resizable(0,0)

# Pantalla de entrada
pantalla = Entry(root, width=22, bg="black", fg="white", borderwidth=0, font=('arial', 18, 'bold'))
pantalla.grid(row=0, column=0, columnspan=4, padx=5, pady=5)

# Funciones
def click_boton(valor):
    pantalla.insert(END, valor)

def borrar():
    pantalla.delete(0, END)

def calcular():
    try:
        expresion = pantalla.get()
        resultado = eval(expresion)
        pantalla.delete(0, END)
        pantalla.insert(0, resultado)
    except:
        pantalla.delete(0, END)
        pantalla.insert(0, "Error")

# Definición de botones
botones = [
    '7', '8', '9', '/',
    '4', '5', '6', '*',
    '1', '2', '3', '-',
    '0', '.', '=', '+'
]

fila = 1
col = 0
for boton_texto in botones:
    if boton_texto == "=":
        Button(root, text=boton_texto, width=9, height=3, command=calcular).grid(row=fila, column=col, padx=1, pady=1)
    else:
        Button(root, text=boton_texto, width=9, height=3, command=lambda val=boton_texto: click_boton(val)).grid(row=fila, column=col, padx=1, pady=1)
    
    col += 1
    if col > 3:
        col = 0
        fila += 1

Button(root, text="Borrar", width=40, height=3, command=borrar).grid(row=5, column=0, columnspan=4, padx=1, pady=1)

root.mainloop()
