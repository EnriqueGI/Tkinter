Diálogos y Ventanas

Cuadros de Diálogo (Messagebox)

Usar from tkinter import messagebox para mostrar información, advertencias o preguntas.


from tkinter import *
from tkinter import messagebox

root = Tk()
root.title("Messagebox")

def mostrar_info():
    messagebox.showinfo("Información", "Esta es una operación exitosa.")

def mostrar_error():
    messagebox.showerror("Error", "Ocurrió un error inesperado.")

def preguntar_salir():
    respuesta = messagebox.askyesno("Confirmar", "¿Estás seguro de que quieres salir?")
    if respuesta:
        root.destroy()

Button(root, text="Mostrar Info", command=mostrar_info).pack(fill=X, padx=20, pady=5)
Button(root, text="Mostrar Error", command=mostrar_error).pack(fill=X, padx=20, pady=5)
Button(root, text="Salir", command=preguntar_salir).pack(fill=X, padx=20, pady=5)

root.mainloop()

Nuevas Ventanas (Toplevel)

Toplevel crea ventanas secundarias asociadas a la principal.


from tkinter import *

root = Tk()
root.title("Ventana Principal")

def abrir_ventana():
    ventana_secundaria = Toplevel(root)
    ventana_secundaria.title("Ventana Secundaria")
    ventana_secundaria.geometry("200x100")
    Label(ventana_secundaria, text="¡Soy una nueva ventana!").pack(pady=20)

Button(root, text="Abrir nueva ventana", command=abrir_ventana).pack(pady=30, padx=30)

root.mainloop()
