Gestores de Geometría

pack()

pack() organiza widgets en bloques, uno debajo del otro o uno al lado del otro.


from tkinter import *
root = Tk()
root.title("Ejemplo de pack()")

Label(root, text="Widget 1 (Top)", bg="red", fg="white").pack(fill=X)
Label(root, text="Widget 2 (Left)", bg="blue", fg="white").pack(side=LEFT, fill=Y)
Label(root, text="Widget 3 (Right)", bg="green", fg="white").pack(side=RIGHT, fill=Y)

root.mainloop()

grid()

grid() organiza widgets en filas y columnas (tabla) para mayor control.


from tkinter import *
root = Tk()
root.title("Ejemplo de grid()")

etiqueta1 = Label(root, text="Fila 0, Columna 0")
etiqueta2 = Label(root, text="Fila 0, Columna 1")
etiqueta3 = Label(root, text="Fila 1, Columna 0")

etiqueta1.grid(row=0, column=0, padx=10, pady=10)
etiqueta2.grid(row=0, column=1, padx=10, pady=10)
etiqueta3.grid(row=1, column=0, padx=10, pady=10)

root.mainloop()

Anclajes (anchor)

Con pack(), podés usar anchor: "n","s","e","w","ne","nw","se","sw","center".


from tkinter import *
root = Tk()
root.geometry("200x200")
root.title("Anclajes")

Label(root, text="Noroeste").pack(anchor="nw")
Label(root, text="Centro").pack(anchor="center")
Label(root, text="Sudeste").pack(anchor="se")

root.mainloop()
